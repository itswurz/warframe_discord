function parseRawTable(md) {
  const rows = [];
  for (const rawLine of md.split('\n')) {
    const line = rawLine.trim();
    if (!line.startsWith('|') || !line.toLowerCase().includes('time:')) continue;
    const cells = line.split('|').map(c=>c.trim()).filter(c=>c.length>0);
    let base_time=480, credits=0;
    const tm = line.match(/time:\s*(\d+(?:\.\d+)?)\s*(day|hour|hr)/i);
    if (tm) { const h=tm[2][0]==='d'?parseFloat(tm[1])*24:parseFloat(tm[1]); base_time=h<=12?300:h<=24?480:720; }
    const ingredients={};
    for (const cell of cells) {
      if (/time:|rush|market|price/i.test(cell)) continue;
      const parts=cell.split('<br>'); if(parts.length<2) continue;
      const amt=parseInt(parts[parts.length-1].replace(/,/g,'')); if(isNaN(amt)||amt<=0) continue;
      const links=[...parts[0].matchAll(/\[([^\]]+)\]\([^)]*\)/g)].map(m=>m[1]).filter(l=>!/^[!h]/.test(l));
      let name=links.length>0?links[links.length-1].trim():'';
      if(!name){const alt=(parts[0].match(/!\[([^\]]+)\]/)||[])[1]||''; if(/credit/i.test(alt)) name='Credits';}
      if(!name) continue;
      if(name==='Credits'){credits=amt;}else{ingredients[name]=amt;}
    }
    if(Object.keys(ingredients).length>0||credits>0) rows.push({credits,base_time,ingredients});
  }
  return rows;
}
